# V50 검증 개요

V50 (Dual-Attention LSTM) 모델의 OOS 성능을 다각도로 검증합니다.
공정 비교 실험(v51b), DM 테스트(v53), 극한 시장 테스트(v59), Attention 분석(v60), 피처 중요도(v61)를 포함합니다.

```{python}
#| echo: false
import json, os

verify_path = '../src/experiments/verification/'

try:
    with open(f'{verify_path}v51b_results.json', 'r') as f:
        v51_data = json.load(f)['exp2_train_scaler']
    with open(f'{verify_path}v53_results.json', 'r') as f:
        v53_data = json.load(f)
    with open(f'{verify_path}v59_results.json', 'r') as f:
        v59_data = json.load(f)
    with open(f'{verify_path}v60_results.json', 'r') as f:
        v60_data = json.load(f)
    with open(f'{verify_path}v61_results.json', 'r') as f:
        v61_data = json.load(f)
except Exception as e:
    print(f"데이터 로드 오류: {e}")
```

## 3. 극한 시장 테스트 (v59)

```{python}
#| echo: false
#| fig-width: 10
#| fig-height: 4

import matplotlib.pyplot as plt

plt.rcParams['font.family'] = 'UnDotum'
plt.rcParams['axes.unicode_minus'] = False

regimes = list(v59_data.keys())
r2s = [v59_data[r]['R2'] for r in regimes]

fig, ax = plt.subplots(figsize=(10, 4))
colors = ['#FF6B6B' if r < 0.4 else '#4ECDC4' for r in r2s]
bars = ax.bar(regimes, r2s, color=colors, edgecolor='black')
ax.set_ylabel('R² Score')
ax.set_title('시장 국면별 V50 성능', fontsize=14, fontweight='bold')
ax.set_ylim(0, 1.0)
ax.grid(True, alpha=0.3, axis='y')

for bar, r2 in zip(bars, r2s):
    ax.text(bar.get_x() + bar.get_width()/2, bar.get_height() + 0.02, f'{r2:.3f}', ha='center', fontweight='bold')

plt.tight_layout()
plt.show()
```

## 4. Attention 분석 (v60)

```{python}
#| echo: false
#| fig-width: 10
#| fig-height: 4

import matplotlib.pyplot as plt
import numpy as np

plt.rcParams['font.family'] = 'UnDotum'
plt.rcParams['axes.unicode_minus'] = False

try:
    weights = v60_data.get('attention_weights', v60_data.get('mean_weights', []))
    if weights:
        fig, ax = plt.subplots(figsize=(10, 4))
        lags = list(range(1, len(weights)+1))
        ax.bar(lags, weights, color='#4299E1', edgecolor='white')
        ax.set_xlabel('Lag (일)')
        ax.set_ylabel('Attention 가중치')
        ax.set_title('V50 Temporal Attention 가중치 분포', fontsize=14, fontweight='bold')
        ax.grid(True, alpha=0.3, axis='y')
        plt.tight_layout()
        plt.show()
    else:
        print("Attention 가중치 데이터 없음")
except Exception as e:
    print(f"시각화 오류: {e}")
```
